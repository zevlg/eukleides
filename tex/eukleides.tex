\let\EukleidesLoaded\endinput

\def\TRS#1#2#3#4{%
    \translate(#1,#2)\rotate{#3}\scale{#4}}

\def\Simple#1#2#3#4{%
  \gsave
    \TRS{#1}{#2}{#3}{#4}
    \moveto(-0.5,-1)
    \lineto(0.5,1)
    \stroke[linewidth=0.125]
  \grestore}

\def\Double#1#2#3#4{%
  \gsave
    \TRS{#1}{#2}{#3}{#4}
    \moveto(-1,-1)
    \lineto(0,1)
    \moveto(0,-1)
    \lineto(1,1)
    \stroke[linewidth=0.125]
  \grestore}

\def\Triple#1#2#3#4{%
  \gsave
    \TRS{#1}{#2}{#3}{#4}
    \moveto(-1.5,-1)
    \lineto(-0.5,1)
    \moveto(-0.5,-1)
    \lineto(0.5,1)
    \moveto(0.5,-1)
    \lineto(1.5,1)
    \stroke[linewidth=0.125]
  \grestore}

\def\Cross#1#2#3#4{%
  \gsave
    \TRS{#1}{#2}{#3}{#4}
    \moveto(1,0)
    \lineto(-1,0)
    \moveto(0,1)
    \lineto(0,-1)
    \stroke[linewidth=0.2]
  \grestore}

\def\Dot#1#2#3{%
  \gsave
    \newpath
    \pscircle(#1,#2){#3}
    \fill[fillstyle=solid,fillcolor=\pslinecolor]
  \grestore}

\def\Dash#1#2#3#4#5{%
  \gsave
    \translate(#1,#2)
    \rotate{#3}
    \moveto(#4,0)
    \lineto(#5,0)
    \stroke
  \grestore}

\def\DoubleDash#1#2#3#4#5#6{%
  \gsave
    \translate(#1,#2)
    \rotate{#5}
    \moveto(#3,0)
    \lineto(#4,0)
    \stroke
    \rotate{#6}
    \moveto(#3,0)
    \lineto(#4,0)
    \stroke
  \grestore}

\def\TripleDash#1#2#3#4#5#6{%
  \gsave
    \translate(#1,#2)
    \rotate{#5}
    \moveto(#3,0)
    \lineto(#4,0)
    \stroke
    \rotate{#6}
    \moveto(#3,0)
    \lineto(#4,0)
    \stroke
    \rotate{#6}
    \moveto(#3,0)
    \lineto(#4,0)
    \stroke
  \grestore}

\def\DoubleArc#1#2#3#4#5#6{%
  \gsave
    \psarc(#1,#2){#3}{#5}{#6}
    \stroke
    \newpath
    \psarc(#1,#2){#4}{#5}{#6}
    \stroke
  \grestore}

\def\TripleArc#1#2#3#4#5#6#7{%
  \gsave
    \psarc(#1,#2){#3}{#6}{#7}
    \stroke
    \newpath
    \psarc(#1,#2){#4}{#6}{#7}
    \stroke
    \newpath
    \psarc(#1,#2){#5}{#6}{#7}
    \stroke
  \grestore}

\def\Right#1#2#3#4#5#6{%
  \gsave
    \moveto(#1,#2)
    \rlineto(#3,#4)
    \rlineto(#5,#6)
    \stroke
  \grestore}
